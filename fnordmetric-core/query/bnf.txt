<query>            ::=
    SELECT
    <select list>
    <from clause>
    [ <where clause> ]
    [ <group by clause> ]
    [ <having clause> ]
    [ <pivot clause> ]

<select list>      ::=
    <asterisk>
  | <select sublist> [ { <comma> <select sublist> }... ]

<select sublist>   ::=
    <derived column>
  | <table name> <period> <asterisk>

<derived column>   ::=
    <value expression> [ [ AS ] <column name> ]

<from clause>      ::=
    FROM <table name> [ { <comma> <table name> }... ]
